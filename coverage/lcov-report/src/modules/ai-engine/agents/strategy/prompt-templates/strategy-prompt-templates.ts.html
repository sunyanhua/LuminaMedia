/**
 * 策划Agent提示词模板库
 * 提供不同营销场景和活动类型的策划方案模板
 */
 
import { AnalysisAgentOutput } from '../../analysis/interfaces/analysis-agent.interface';
import {
  EventInfo,
  HolidayInfo,
  BudgetInfo,
  DateRange,
} from '../interfaces/strategy-agent.interface';
 
export interface StrategyPromptTemplate {
  /** 模板ID */
  id: string;
  /** 模板名称 */
  name: string;
  /** 适用营销场景 */
  scenarios: string[];
  /** 活动类型 */
  campaignTypes: string[];
  /** 策划深度级别 (1-5, 5为最深) */
  depthLevel: number;
  /** 提示词模板函数 */
  generatePrompt: (params: StrategyPromptParams) => string;
}
 
export interface StrategyPromptParams {
  /** 分析阶段结果 */
  analysisResults: AnalysisAgentOutput;
  /** 时事热点 */
  currentEvents: EventInfo[];
  /** 节假日信息 */
  holidays: HolidayInfo[];
  /** 预算约束 */
  budgetConstraints: BudgetInfo;
  /** 时间范围 */
  timeline: DateRange;
  /** 行业背景 */
  industryContext: string;
  /** 自定义参数 */
  customParams?: Record<string, any>;
}

### 3. 渠道计划

- **渠道策略**：每个渠道的具体执行策略

- **预算分配**：各渠道的预算分配建议

/**
 * 生成分析结果摘要
 */
function generateAnalysisSummary(analysis: AnalysisAgentOutput): string {
  const {
    marketInsights,
    targetAudience,
    competitorAnalysis,
    recommendations,
  } = analysis;
 
  return `### 市场洞察摘要
- 主要趋势：${marketInsights.trends.slice(0, 3).join('、')}
- 关键机会：${marketInsights.opportunities.slice(0, 3).join('、')}
- 主要威胁：${marketInsights.threats.slice(0, 2).join('、')}
 
### 目标客群摘要
- 细分群体：${targetAudience.segments
    .slice(0, 3)
