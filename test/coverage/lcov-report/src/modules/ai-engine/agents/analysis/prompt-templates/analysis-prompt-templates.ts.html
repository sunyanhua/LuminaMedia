/**
 * 分析Agent提示词模板库
 * 提供不同行业和分析深度的提示词模板
 */
 
export interface AnalysisPromptTemplate {
  /** 模板ID */
  id: string;
  /** 模板名称 */
  name: string;
  /** 适用行业 */
  industries: string[];
  /** 分析深度级别 (1-5, 5为最深) */
  depthLevel: number;
  /** 提示词模板函数 */
  generatePrompt: (params: AnalysisPromptParams) => string;
}
 
export interface AnalysisPromptParams {
  /** 行业背景 */
  industryContext: string;
  /** 业务目标 */
  businessGoals: string[];
  /** 用户画像数据摘要 */
  profileSummary: string;
  /** 知识库上下文 */
  knowledgeContext: string[];
  /** 自定义参数 */
  customParams?: Record<string, any>;
}
 
/**
 * 标准分析模板 - 适用于大多数行业
 */
export const standardAnalysisTemplate: AnalysisPromptTemplate = {
  id: 'standard',
  name: '标准市场分析模板',
  industries: ['通用'],
  depthLevel: 3,
  generatePrompt: (params: AnalysisPromptParams) => {
    const { industryContext, businessGoals, profileSummary, knowledgeContext } =
      params;
 
    const knowledgeText =
      knowledgeContext.length > 0
        ? `相关知识库内容：\n${knowledgeContext.slice(0, 3).join('\n')}`
        : '暂无相关知识库内容。';
 
    return `你是一位资深的市场营销分析专家。请基于以下信息进行全面的市场分析：
 
## 业务背景
- 行业：${industryContext}
- 业务目标：${businessGoals.join('、')}
 
## 目标客群画像
${profileSummary}
 
## 知识库参考
${knowledgeText}
 
## 分析要求
请提供完整的市场分析报告，包括以下四个部分：
 
### 1. 市场洞察
- **市场趋势**：识别当前行业的主要发展趋势（至少3条）
- **市场机会**：发现可把握的市场机会（至少3个）
- **市场威胁**：分析可能面临的市场威胁（至少2个）
 
### 2. 目标客群分析
- **客群分群**：基于提供的用户画像数据，将目标客群划分为3-5个细分群体
- **典型用户画像**：创建一个最具代表性的典型用户画像
- **规模预估**：基于行业数据和样本特征，预估目标客群总体规模
 
### 3. 竞品分析
- **主要竞争对手**：识别行业内的主要竞争对手（3-5个）
- **竞争优势**：分析我们相比竞争对手的潜在优势（至少3个）
- **差距分析**：识别我们与竞争对手之间的差距（至少2个）
 
### 4. 初步建议
基于以上分析，提出针对性的营销建议（至少5条具体可执行的建议）
 
请以结构化JSON格式输出分析结果。`;
  },
};

/**
 * 模板选择器
 */
export function selectTemplate(
  industry: string,
  depthRequired: number = 3,
): AnalysisPromptTemplate {
  const industryLower = industry.toLowerCase();
 
  // 按行业匹配
  if (
    ['电商', '零售', '消费品', 'ecommerce'].some((keyword) =>
      industryLower.includes(keyword),
    )
  ) {
    return ecommerceAnalysisTemplate;
  }
 
  if (
    ['企业服务', 'saas', '软件', '咨询', 'b2b'].some((keyword) =>
      industryLower.includes(keyword),
    )
  ) {
    return b2bAnalysisTemplate;
  }
 
  if (
    ['金融', '医疗', '教育'].some((keyword) => industryLower.includes(keyword))
  ) {
    return deepIndustryAnalysisTemplate;
  }
 
  // 按深度需求匹配
  if (depthRequired >= 4) {
    return deepIndustryAnalysisTemplate;
  } else if (depthRequired <= 2) {
    return quickScanTemplate;
  }
 
  // 默认返回标准模板
  return standardAnalysisTemplate;
}
